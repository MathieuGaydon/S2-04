-- **************************************************************************** Définition des données


PROMPT "Définition des données";

CREATE TABLE Organisme(
  refOrganisme NUMBER,
  typeOrganisme VARCHAR(50) NOT NULL,
  raisonSociale VARCHAR(50) NOT NULL,
  numeroSIRET NUMBER NOT NULL,
  PRIMARY KEY(refOrganisme)
);

CREATE TABLE Organisation(
  numero NUMBER,
  PRIMARY KEY(numero)
);

CREATE TABLE Date_(
  date_ DATE,
  PRIMARY KEY(date_)
);

CREATE TABLE Rang(
  rang VARCHAR(50),
  PRIMARY KEY(rang)
);

CREATE TABLE Titre(
  titre VARCHAR(50),
  PRIMARY KEY(titre)
);

CREATE TABLE Dignite(
  dignite VARCHAR(50),
  PRIMARY KEY(dignite)
);

CREATE TABLE Repas(
  idRepas NUMBER,
  nomRepas VARCHAR(50) NOT NULL,
  PRIMARY KEY(idRepas)
);


CREATE TABLE Aliment(
  idAliment NUMBER,
  nomAliment VARCHAR(50) NOT NULL,
  PRIMARY KEY(idAliment)
);

CREATE TABLE Legume(
  idLegume NUMBER,
  nomLegume VARCHAR(50),
  PRIMARY KEY(idLegume)
);

CREATE TABLE Sauce(
  idSauce NUMBER,
  nomSauce VARCHAR(50) NOT NULL,
  PRIMARY KEY(idSauce)
);

CREATE TABLE Ingredient(
  idIngredient NUMBER,
  nomIngredient VARCHAR(50),
  PRIMARY KEY(idIngredient)
);


CREATE TABLE Machine(
  idMachine NUMBER,
  nomMachine VARCHAR(50),
  PRIMARY KEY(idMachine)
);

CREATE TABLE Grade(
  grade VARCHAR(50),
  PRIMARY KEY(grade)
);

CREATE TABLE GradeSuperieur(
  gradeSup VARCHAR(50),
  PRIMARY KEY(gradeSup)
);

CREATE TABLE Tenrac(
  numero NUMBER,
  refOrganisme NUMBER,
  codeMembre NUMBER,
  nom VARCHAR(50) NOT NULL,
  prenom VARCHAR(50) NOT NULL,
  courriel VARCHAR(50) NOT NULL,
  numeroTel VARCHAR(50) NOT NULL,
  adressePostale VARCHAR(70) NOT NULL,
  gradeSup VARCHAR(50),
  grade VARCHAR(50),
  dignite VARCHAR(50),
  titre VARCHAR(50),
  rang VARCHAR(50),
  PRIMARY KEY(numero, refOrganisme, codeMembre),
  FOREIGN KEY(numero) REFERENCES Organisation(numero),
  FOREIGN KEY(refOrganisme) REFERENCES Organisme(refOrganisme),
  FOREIGN KEY(gradeSup) REFERENCES GradeSuperieur(gradeSup),
  FOREIGN KEY(grade) REFERENCES Grade(grade),
  FOREIGN KEY(dignite) REFERENCES Dignite(dignite),
  FOREIGN KEY(titre) REFERENCES Titre(titre),
  FOREIGN KEY(rang) REFERENCES Rang(rang)
);

CREATE TABLE Ordre(
  numero NUMBER,
  nomOrdre VARCHAR(50) NOT NULL,
  PRIMARY KEY(numero),
  FOREIGN KEY(numero) REFERENCES Organisation(numero)
);

CREATE TABLE Partenaire(
  adresse VARCHAR(50),
  numero NUMBER NOT NULL,
  PRIMARY KEY(adresse),
  FOREIGN KEY(numero) REFERENCES Ordre(numero)
);

CREATE TABLE Reunion(
  idReunion NUMBER,
  PRIMARY KEY(idReunion)
);

CREATE TABLE Plat(
  idPlat NUMBER,
  nomPlat VARCHAR(50) NOT NULL,
  idLegume NUMBER,
  PRIMARY KEY(idPlat),
  FOREIGN KEY(idLegume) REFERENCES Legume(idLegume)
);

CREATE TABLE Entretient(
  idEntretient NUMBER,
  periodeEntretient DATE NOT NULL,
  typeEntretient VARCHAR(50) NOT NULL,
  numero NUMBER,
  PRIMARY KEY(idEntretient),
  FOREIGN KEY(numero) REFERENCES Ordre(numero)
);

CREATE TABLE Club(
  numero_1 NUMBER,
  nomClub VARCHAR(50) NOT NULL,
  numero NUMBER,
  PRIMARY KEY(numero_1),
  FOREIGN KEY(numero_1) REFERENCES Organisation(numero),
  FOREIGN KEY(numero) REFERENCES Ordre(numero)
);


CREATE TABLE Modele(
  idModele NUMBER,
  nomModele VARCHAR(50),
  idEntretient NUMBER NOT NULL,
  PRIMARY KEY(idModele),
  FOREIGN KEY(idEntretient) REFERENCES Entretient(idEntretient)
);


CREATE TABLE SeGroupe(
  adresse VARCHAR(50),
  date_ DATE,
  idReunion NUMBER,
  PRIMARY KEY(adresse, date_, idReunion),
  FOREIGN KEY(adresse) REFERENCES Partenaire(adresse),
  FOREIGN KEY(date_) REFERENCES Date_(date_),
  FOREIGN KEY(idReunion) REFERENCES Reunion(idReunion)
);


CREATE TABLE Rejoint(
  numero NUMBER,
  refOrganisme NUMBER,
  codeMembre NUMBER,
  idReunion NUMBER,
  PRIMARY KEY(numero, refOrganisme, codeMembre, idReunion),
  FOREIGN KEY(numero, refOrganisme, codeMembre) REFERENCES Tenrac(numero, refOrganisme, codeMembre),
  FOREIGN KEY(idReunion) REFERENCES Reunion(idReunion)
);


CREATE TABLE Compose(
  idRepas NUMBER,
  idPlat NUMBER,
  PRIMARY KEY(idRepas, idPlat),
  FOREIGN KEY(idRepas) REFERENCES Repas(idRepas),
  FOREIGN KEY(idPlat) REFERENCES Plat(idPlat)
);


CREATE TABLE Constitue(
  idPlat NUMBER,
  idAliment NUMBER,
  PRIMARY KEY(idPlat, idAliment),
  FOREIGN KEY(idPlat) REFERENCES Plat(idPlat),
  FOREIGN KEY(idAliment) REFERENCES Aliment(idAliment)
);


CREATE TABLE Accompagne(
  idPlat NUMBER,
  idSauce NUMBER,
  PRIMARY KEY(idPlat, idSauce),
  FOREIGN KEY(idPlat) REFERENCES Plat(idPlat),
  FOREIGN KEY(idSauce) REFERENCES Sauce(idSauce)
);


CREATE TABLE forme(
  idSauce NUMBER,
  idIngredient NUMBER,
  PRIMARY KEY(idSauce, idIngredient),
  FOREIGN KEY(idSauce) REFERENCES Sauce(idSauce),
  FOREIGN KEY(idIngredient) REFERENCES Ingredient(idIngredient)
);


CREATE TABLE Mange(
  date_ DATE,
  idReunion NUMBER,
  idRepas NUMBER,
  PRIMARY KEY(date_, idReunion, idRepas),
  FOREIGN KEY(date_) REFERENCES Date_(date_),
  FOREIGN KEY(idReunion) REFERENCES Reunion(idReunion),
  FOREIGN KEY(idRepas) REFERENCES Repas(idRepas)
);


CREATE TABLE Caracterise(
  idMachine NUMBER,
  idModele NUMBER,
  PRIMARY KEY(idMachine, idModele),
  FOREIGN KEY(idMachine) REFERENCES Machine(idMachine),
  FOREIGN KEY(idModele) REFERENCES Modele(idModele)
);


CREATE TABLE Realise(
  numero NUMBER,
  refOrganisme NUMBER,
  codeMembre NUMBER,
  idMachine NUMBER,
  idEntretient NUMBER,
  PRIMARY KEY(numero, refOrganisme, codeMembre, idMachine, idEntretient),
  FOREIGN KEY(numero, refOrganisme, codeMembre) REFERENCES Tenrac(numero, refOrganisme, codeMembre),
  FOREIGN KEY(idMachine) REFERENCES Machine(idMachine),
  FOREIGN KEY(idEntretient) REFERENCES Entretient(idEntretient)
);


CREATE TABLE UtilisationOfficielle(
  idRepas NUMBER,
  idMachine NUMBER,
  PRIMARY KEY(idRepas, idMachine),
  FOREIGN KEY(idRepas) REFERENCES Repas(idRepas),
  FOREIGN KEY(idMachine) REFERENCES Machine(idMachine)
);


CREATE TABLE Intolerance(
  numero NUMBER,
  refOrganisme NUMBER,
  codeMembre NUMBER,
  idLegume NUMBER,
  PRIMARY KEY(numero, refOrganisme, codeMembre, idLegume),
  FOREIGN KEY(numero, refOrganisme, codeMembre) REFERENCES Tenrac(numero, refOrganisme, codeMembre),
  FOREIGN KEY(idLegume) REFERENCES Legume(idLegume)
);


CREATE TABLE Enregistre1(
  numero NUMBER,
  idEntretient NUMBER,
  PRIMARY KEY(numero, idEntretient),
  FOREIGN KEY(numero) REFERENCES Club(numero_1),
  FOREIGN KEY(idEntretient) REFERENCES Entretient(idEntretient)
);


CREATE TABLE Dirige(
  numero NUMBER,
  refOrganisme NUMBER,
  codeMembre NUMBER,
  idReunion NUMBER,
  PRIMARY KEY(idReunion, numero , refOrganisme, codeMembre),
  FOREIGN KEY(idReunion) REFERENCES Reunion(idReunion),
  FOREIGN KEY(numero, refOrganisme, codeMembre) REFERENCES Tenrac (numero, refOrganisme, codeMembre)
);
